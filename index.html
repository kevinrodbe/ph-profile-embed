Product Hunt Profile Widget

Embed your Product Hunt profile on any website.

View on GitHub

Instructions

Enter your username below, and then copy-paste the code shown onto your website.



<script type="text/javascript" id="embeddable-ph-{{username}}" >
  (function() {
    function async_load(){
      var s = document.createElement('script');
      s.type = 'text/javascript';
      s.async = true;
      var theUrl = 'http://ph-profile-embed.herokuapp.com/user/{{username}}';
      s.src = theUrl + ( theUrl.indexOf("?") >= 0 ? "&" : "?") + 'ref=' + encodeURIComponent(window.location.href);
      var embedder = document.getElementById('embeddable-ph-{{username}}');
      embedder.parentNode.insertBefore(s, embedder);
    }
    if (window.attachEvent)
      window.attachEvent('onload', async_load);
    else
      window.addEventListener('load', async_load, false);
  })();
</script>
